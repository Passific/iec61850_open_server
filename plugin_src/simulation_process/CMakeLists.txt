
set(simulation_SRCS
   simulation.c
)

    ${CMAKE_CURRENT_SOURCE_DIR}/../../inc
    ${IEC61850_INCLUDE_DIR}
)

    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}"
)
